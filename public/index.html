$(document).ready(function(){
    var scrollBar=$('#switch');
    var gauge1=$('[data-role=gauge]').gauge({
        padding:20,
        radius:scrollBar.width()/2-50,
        ticksRadius:scrollBar.width()/2-40,
        ticksLength:10,
        ticksLabelRadius:scrollBar.width()/2-20,
        arrowAxisRadius:20,
        arrowRadius:scrollBar.width()/2-50
    });
    scrollBar.scroll(function(e){
        gauge1.gauge("setValue",($(this).scrollTop()/10));
    }).children('div').css({
        'height':'30px',
        'width':scrollBar.width()*11+'px'
    });
});
