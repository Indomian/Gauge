window.onload=function(){
    var scrollBar=document.getElementById('switch');
    var gauge1=new Gauge(document.getElementById('gauge'),{
        padding:20,
        radius:scrollBar.offsetWidth/2-50,
        ticksRadius:scrollBar.offsetWidth/2-40,
        ticksLength:10,
        ticksLabelRadius:scrollBar.offsetWidth/2-20,
        arrowAxisRadius:20,
        arrowRadius:scrollBar.offsetWidth/2-50
    });
    scrollBar.onscroll=function(e){
        gauge1.setValue(this.scrollLeft/this.offsetWidth*10);
    };
    scrollBar.children[0].setAttribute('style','height:30px;width:'+(scrollBar.offsetWidth*11)+'px;');
}
